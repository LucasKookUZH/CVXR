function (X, P) 
{
    if (is.matrix(P) && (is.vector(X) || ncol(X) == 1)) {
        invP <- as.matrix(base::solve(P))
        return(QuadForm(x = X, P = (invP + t(Conj(invP)))/2))
    }
    else return(MatrixFrac(X = X, P = P))
}
